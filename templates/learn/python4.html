การเขียนโปรแกรมเชิงวัตถุ (OOP)

คลาส (Classes) และวัตถุ (Objects)


# การสร้างคลาส
class Person:
    # คลาสแอททริบิวต์ (ใช้ร่วมกันโดยทุกอินสแตนซ์)
    species = "Human"

    # เมธอด constructor
    def __init__(self, name, age):
        # อินสแตนซ์แอททริบิวต์ (เฉพาะแต่ละอินสแตนซ์)
        self.name = name
        self.age = age

    # เมธอดของคลาส
    def greet(self):
        return f"Hello, my name is {self.name}"

    def is_adult(self):
        return self.age >= 18

การสร้างและใช้งานวัตถุ (Objects)


        # สร้างอินสแตนซ์ของคลาส
        person1 = Person("John", 25)
        person2 = Person("Alice", 17)
        
        # เข้าถึงแอททริบิวต์
        print(person1.name)        # "John"
        print(person2.age)         # 17
        print(person1.species)     # "Human"
        
        # เรียกใช้เมธอด
        print(person1.greet())     # "Hello, my name is John"
        print(person2.is_adult())  # False
