โมดูลและแพ็คเกจ

การใช้งานโมดูล (Modules)

โมดูลคือไฟล์ Python ที่มีฟังก์ชัน, คลาส, และตัวแปรที่คุณสามารถนำมาใช้ในไฟล์อื่นได้

การ Import โมดูลมาตรฐาน


# Import ทั้งโมดูล
import math
print(math.sqrt(16))  # 4.0

# Import เฉพาะบางฟังก์ชัน
from math import sqrt, pi
print(sqrt(16))  # 4.0
print(pi)        # 3.141592...

# Import ทั้งหมดและใช้โดยตรง (ไม่แนะนำให้ใช้กับโมดูลขนาดใหญ่)
from math import *
print(sqrt(16))   # 4.0
print(cos(0))     # 1.0

# Import และตั้งชื่อย่อ
import math as m
print(m.sqrt(16))  # 4.0

โมดูลมาตรฐานที่ใช้บ่อย

1. math: ฟังก์ชันทางคณิตศาสตร์


import math
print(math.sqrt(16))       # รากที่สอง: 4.0
print(math.factorial(5))   # 5!: 120
print(math.pi)             # ค่า π: 3.141592...

2. random: การสุ่มค่า


import random
print(random.randint(1, 10))            # สุ่มเลข 1-10
print(random.choice(["A", "B", "C"]))   # สุ่มเลือกจากลิสต์

3. datetime: การทำงานกับวันที่และเวลา


from datetime import datetime, timedelta

now = datetime.now()
print(now)                          # เวลาปัจจุบัน
tomorrow = now + timedelta(days=1)
print(tomorrow)                     # วันพรุ่งนี้ในเวลาเดียวกัน
print(now.strftime("%d/%m/%Y"))     # รูปแบบวันที่: "DD/MM/YYYY"

4. os: การทำงานกับระบบปฏิบัติการ


import os
print(os.getcwd())              # แสดงไดเรกทอรีปัจจุบัน
os.mkdir("new_folder")          # สร้างโฟลเดอร์ใหม่
files = os.listdir(".")         # แสดงรายการไฟล์ในโฟลเดอร์ปัจจุบัน

# mymodule.py
def greet(name):
    return f"Hello, {name}!"

def add(a, b):
    return a + b

PI = 3.14159

# main.py

แพ็คเกจ (Packages)

แพ็คเกจคือไดเรกทอรีที่มีไฟล์ __init__.py และอาจมีโมดูลหรือแพ็คเกจย่อยอื่นๆ

โครงสร้างแพ็คเกจ

mypackage/
    __init__.py
    module1.py
    module2.py
    subpackage/
        __init__.py
        module3.py

การใช้งานแพ็คเกจ


# Import โมดูลจากแพ็คเกจ
import mypackage.module1
mypackage.module1.some_function()

# Import ฟังก์ชันโดยตรง
from mypackage.module2 import another_function
another_function()

# Import จากแพ็คเกจย่อย
from mypackage.subpackage.module3 import special_function
special_function()

การติดตั้งแพ็คเกจผ่าน pip

pip คือ package manager สำหรับ Python ที่ใช้ติดตั้งแพ็คเกจจาก Python Package Index (PyPI)


# ติดตั้งแพ็คเกจ
pip install package_name

# ติดตั้งแพ็คเกจเวอร์ชันเฉพาะ
pip install package_name==1.0.0

# อัปเกรดแพ็คเกจ
pip install --upgrade package_name

# ถอนการติดตั้งแพ็คเกจ
pip uninstall package_name

# แสดงรายการแพ็คเกจที่ติดตั้ง
pip list

แพ็คเกจยอดนิยม
