พารามิเตอร์และการ Return


# ฟังก์ชันที่มีพารามิเตอร์หลายตัว
def add(a, b):
    return a + b

# ฟังก์ชันที่ไม่ return อะไร
def print_info(name, age):
    print(f"Name: {name}, Age: {age}")

# ฟังก์ชันที่ return หลายค่า
def get_person_details():
    name = "John"
    age = 25
    job = "Developer"
    return name, age, job  # Return เป็น tuple

# การเรียกใช้
result = add(3, 5)
print(result)  # 8

print_info("Alice", 30)  # Name: Alice, Age: 30

name, age, job = get_person_details()
print(f"{name} is a {age}-year-old {job}")

Default Parameters และ Named Arguments


# พารามิเตอร์ที่มีค่าเริ่มต้น
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

# การเรียกใช้
print(greet("John"))  # "Hello, John!"
print(greet("John", "Hi"))  # "Hi, John!"

# Named arguments
print(greet(greeting="Hey", name="Alice"))  # "Hey, Alice!"

*args และ **kwargs


# *args รับพารามิเตอร์ไม่จำกัดจำนวน
def sum_all(*args):
    return sum(args)

print(sum_all(1, 2, 3))  # 6
print(sum_all(1, 2, 3, 4, 5))  # 15

# **kwargs รับพารามิเตอร์แบบ key-value ไม่จำกัดจำนวน
def print_person_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_person_info(name="John", age=25, job="Developer")

Lambda Functions

Lambda คือฟังก์ชันแบบสั้นๆ ที่เขียนได้ในบรรทัดเดียว:


# รูปแบบ: lambda arguments: expression

# ฟังก์ชันแบบปกติ
def multiply(x, y):
    return x * y

# ฟังก์ชันแบบ lambda ที่ทำงานเหมือนกัน
multiply_lambda = lambda x, y: x * y

print(multiply(3, 4))       # 12
print(multiply_lambda(3, 4))  # 12

Scope ของตัวแปร (Variable Scope)


# ตัวแปรแบบ global
x = 10

def my_function():
    # ตัวแปรแบบ local
    y = 5
    print(x)  # สามารถเข้าถึงตัวแปร global ได้
    print(y)  # เข้าถึงตัวแปร local ได้

my_function()
print(x)  # ยังคงเข้าถึงตัวแปร global ได้
# print(y)  # Error: ไม่สามารถเข้าถึงตัวแปร local นอกฟังก์ชันได้

def change_global():
    global x  # ประกาศว่าจะใช้ตัวแปร global x
    x = 20    # เปลี่ยนค่าตัวแปร global

change_global()
print(x)  # 20
