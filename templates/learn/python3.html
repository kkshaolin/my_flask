พารามิเตอร์และการ Return

```python # ฟังก์ชันที่มีพารามิเตอร์หลายตัว def add(a, b): return a + b # ฟังก์ชันที่ไม่ return อะไร def print_info(name, age): print(f"Name: {name}, Age: {age}") # ฟังก์ชันที่ return หลายค่า def get_person_details(): name = "John" age = 25 job = "Developer" return name, age, job # Return เป็น tuple # การเรียกใช้ result = add(3, 5) print(result) # 8 print_info("Alice", 30) # Name: Alice, Age: 30 name, age, job = get_person_details() print(f"{name} is a {age}-year-old {job}")
